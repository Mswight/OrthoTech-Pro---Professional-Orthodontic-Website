API Integration Example

patient-api.js 
// OrthoTech Pro Patient API v2.0
import { SecureAuth, EncryptionService } from '@orthotech/security';

class PatientPortal {
  constructor() {
    this.auth = new SecureAuth({
      encryption: 'AES-256-GCM',
      tokenExpiry: 3600,
      mfa: true
    });
  }

  async authenticate(credentials) {
    const token = await this.auth.verifyBiometric(credentials);
    
    if (token.verified) {
      return await this.loadPatientData(token.userId);
    }
    
    throw new AuthError('Authentication failed');
  }

  async getTreatmentProgress(patientId) {
    const encrypted = await fetch(`/api/progress/${patientId}`);
    return EncryptionService.decrypt(encrypted);
  }
}

export default PatientPortal;
